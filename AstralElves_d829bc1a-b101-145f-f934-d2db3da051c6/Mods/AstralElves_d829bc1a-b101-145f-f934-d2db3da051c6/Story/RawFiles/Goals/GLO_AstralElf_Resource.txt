ApplyStatus(_Character, "STARLIGHT_STEP_LONGREST", 1.0);

IF
StatusApplied(_Character, "END_RESTORATION", _, _)
THEN
ApplyStatus(_Character, "STARLIGHT_STEP_LONGREST", 1.0);

IF
StatusApplied(_Character, "ALCH_POTION_REST_SLEEP_GREATER_RESTORATION", _, _)
THEN
ApplyStatus(_Character, "STARLIGHT_STEP_LONGREST", 1.0);

IF
CastSpell(_Character, "Shout_DivineIntervention_Healing", _, _, _)
THEN
ApplyStatus(_Character, "STARLIGHT_STEP_LONGREST", 1.0);

IF
UsingSpellOnTarget(_Caster, _Target, "Shout_DivineIntervention_Healing", _, _, _)
AND
DB_PartyMembers((CHARACTER)_Target)
THEN
ApplyStatus(_Target, "STARLIGHT_STEP_LONGREST", 1.0);
